## REPLACEMENT STRATEGY (in priority order)

### PRIMARY: Groq Whisper API (free tier, very fast)
```bash
npm install groq-sdk
```
```javascript
// backend/src/services/transcriptionService.js

import Groq from "groq-sdk";

const groq = new Groq({ apiKey: process.env.GROQ_API_KEY });

async function transcribeAudio(audioPath) {
  const file = fs.createReadStream(audioPath);
  
  const transcription = await groq.audio.transcriptions.create({
    file,
    model: "whisper-large-v3",      // same Whisper model, free on Groq
    response_format: "verbose_json", // returns segments with timestamps
    timestamp_granularities: ["segment"]
  });

  // Returns same format as OpenAI Whisper — zero code change needed elsewhere
  return transcription.segments; // [{ start, end, text }]
}
```

**Why Groq:** Free tier = 7,200 minutes of audio/day. No credit card needed.
Get key at: console.groq.com (free account)

### FALLBACK: Local Whisper via Python (100% offline, no API key)

If Groq fails or quota exceeded, auto-fallback to local:
```javascript
async function transcribeLocal(audioPath) {
  return new Promise((resolve, reject) => {
    // Runs: python3 -m whisper audio.wav --model small --output_format json
    const py = spawn('python3', [
      '-m', 'whisper', audioPath,
      '--model', 'small',         // 'small' = fast, good quality, ~244MB
      '--output_format', 'json',
      '--output_dir', '/tmp/',
      '--language', 'auto'        // auto-detect language
    ]);

    py.on('close', (code) => {
      if (code !== 0) return reject(new Error('Whisper local failed'));
      const result = JSON.parse(
        fs.readFileSync(audioPath.replace('.wav', '.json'))
      );
      resolve(result.segments); // same format [{ start, end, text }]
    });
  });
}
```
